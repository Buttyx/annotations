 * A module containing the ACCESS values enum.
 * @module ACCESS

    "use strict";

    /**
     * Enum for access to ressource
     *
     * @readonly
     * @enum {number}
     */
    return {
        /** Private = 0*/
        PRIVATE: 0,
        /** Public = 1*/
        PUBLIC: 1,
        /** Group = 2*/
        GROUP: 2
    };
});
